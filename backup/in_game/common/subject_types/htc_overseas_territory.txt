overseas_territory = {
	subject_pays = subject_pays_march
	type = location

	use_overlord_map_color = yes
	use_overlord_map_name = yes
	only_overlord_culture = yes
	only_overlord_court_language = yes
	use_overlord_laws = yes

	level = 2

	creation_visible = {
		is_ai = no
	}

	subject_creation_enabled = {
		is_ai = no
	}

	release_country_enabled = {
		always = no
	}
	
	visible_through_diplomacy = {
		always = no
	}

	visible_through_treaty = {
		always = no
	}
	
	enabled_through_diplomacy = {
		always = no
	}

	# root = subject type, scope:actor = caller, scope:recipient = callee, scope:target (optional) = against who
	join_offensive_wars_can_call = {
		scope:recipient = {
			is_subject_of = scope:actor
		}
	}
	join_defensive_wars_can_call = {
		scope:actor = {
			is_subject_of = scope:recipient
		}
	}
	allow_declaring_wars = {
		always = yes
	}
	
	diplomatic_capacity_cost_scale = 0.1
	
	strength_vs_overlord = -0.5
	
	can_be_annexed = yes
	annexation_speed = 1
	annexation_min_years_before = 20
	annexation_min_opinion = 150
	annexation_stall_opinion = 125

	has_overlords_ruler = yes
	overlord_can_cancel = yes	

	allow_subjects = no
	overlord_share_exploration = yes
	can_overlord_build_roads = yes
	can_overlord_build_buildings = yes
	can_overlord_build_rgos = yes
	can_overlord_build_ships = yes
	has_limited_diplomacy = yes
	fleet_basing_rights = yes
	food_access = yes

	can_change_rank = yes
	can_change_heir_selection = no
	great_power_score_transfer = 0.75 # the amount of Great Power Score that is transferred from the Subject to the Overlord
	
	institution_spread_to_overlord = monthly_institution_spread_mild
	institution_spread_to_subject = monthly_institution_spread_mild
	
	overlord_modifier = {
		monthly_towards_outward = societal_value_minor_monthly_move
		global_merchant_power = 0.05
		trade_sea_movement_cost_modifier = -0.15
	}
	
	subject_modifier = {
		global_merchant_power = 0.15
		global_merchant_capacity_modifier = 0.15
		global_maritime_presence_modifier = 0.5
		trade_sea_movement_cost_modifier = -0.15
		loyalty_to_overlord = 40
	}

	on_enable = {
		ordered_owned_location = {
			max = 1
			order_by = best_capital_for_colony
			save_scope_as = new_fort
		}
		scope:new_fort = {
			if = {
				limit = { 
					not = { has_building = building_type:overseas_fort }
				}
				construct_building = {
					building_type = building_type:overseas_fort
					cost_multiplier = 0
				}
			}
		}
		set_capital = scope:new_fort
	}
	on_disable = {}

	diplo_chance_accept_subject = {
		base = -1000
	}
	diplo_chance_accept_overlord = {
		base = -1000
	}
}
