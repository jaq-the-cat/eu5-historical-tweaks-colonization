estabilish_trade_post = {
	ability = dip
	potential = {
		is_rebel_country = no
		num_locations > 0
	}
	allow = {
		or = {
			is_ai = no
			# renaissance
			has_advance = por_pursuing_waves
			has_advance = route_to_the_indies_advance
			# discovery
			has_advance = coastal_trade_routes_advance
			# reformation
			has_advance = global_trade_routes_advance
		}
	}

	select_trigger = {
		looking_for_a = province_definition
		interaction_source_list = {
			every_province_definition = {
				limit = {
					any_location_in_province_definition = {
						has_owner = no
						is_ownable = yes
						is_coastal = yes
					}
					region = {
						not = {
							any_location_in_region = {
								owner ?= scope:actor
							}
						}
					}
				}
				add_to_list = source
			}
		}
		target_flag = target
		name = "estabilish_trade_post_select_province_definition"
		none_available_msg_key = "estabilish_trade_post_no_provinces"
		column = { data = name }
		column = { data = population }
		visible = {
			not = { cabinet_already_performing_same_task_on_target = { cabinet_action = estabilish_trade_post interaction_target = target } }
		}
		enabled = {}
	}

	select_trigger = {
		looking_for_a = location
		source = target
		target_flag = target_1
		name = "estabilish_trade_post_select_location"
		none_available_msg_key = "estabilish_trade_post_no_locations"
		column = { data = name }
		column = { data = population }
		visible = {
			not = { cabinet_already_performing_same_task_on_target = { cabinet_action = estabilish_trade_post interaction_target = target_1 } }
			has_owner = no
			is_ownable = yes
			is_coastal = yes
		}
		enabled = {}
	}
	
	years = 1

	on_fully_activated = {
		scope:target_1 = {
			change_location_owner = scope:actor
			change_integration_level = colonized
		}
	}

	on_activate = {}
	on_deactivate = {}

	is_finished = {
		or = {
			# finished implementing!
			cabinet_action:estabilish_trade_post = {
				is_fully_implemented_in = root
			}
			# we already own a province in that region
			scope:target = {
				region = {
					any_location_in_region = {
						owner ?= scope:actor
					}
				}
			}
		}
	}
	progress = {
		cabinet_action:estabilish_trade_post = {
			value = "implementation_progress_percentage(scope:actor)"
		}
	}

	ai_will_do = {
		if = {
			limit = {
				scope:actor = {
					or = {
						# portugal and castille explore pre-1500
						and = {
							current_year < 1500
							or = {
								has_or_had_tag = POR
								has_or_had_tag = CAS
								has_or_had_tag = SPA
							}
						}
						# france and england join in to explore post-1500
						and = {
							current_year >= 1500
							or = {
								has_or_had_tag = POR
								has_or_had_tag = CAS
								has_or_had_tag = SPA
								has_or_had_tag = FRA
								has_or_had_tag = ENG
								has_or_had_tag = GBR
							}
						}
					}
				}
				scope:target = {
					or = {
						continent = continent:africa
						region = region:indonesia_region
						region = region:melanesia_region
						region = region:new_zealand_region
						sub_continent = sub_continent:pacific_islands
					}
					region = {
						any_location_in_region = {
							not = { owner ?= scope:actor }
						}
					}
				}
			}
			add = 1000
		}
		else = {
			add = -1000
		}
	}
}
