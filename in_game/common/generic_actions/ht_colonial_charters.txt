# scope:actor = country, scope:recipient = country
 
REPLACE:create_colonial_charter = {
	type = owncountry
	
	potential = {
		scope:actor = {
			num_provinces > 0
			is_rebel_country = no
			NOT = { country_type = building }
			modifier:can_colonize = yes
			OR = {
				is_ai = no 
				ai_country_should_colonize = yes
			}
		}
	}

	allow = {
		scope:actor = {
			total_population >= 5
		}
	}

	ai_tick = monthly
	ai_tick_frequency = 6
	automation_tick = monthly
	automation_tick_frequency = 1

	player_automated_category = colonies

	
	price = price:create_colonial_charter
	
	select_trigger = {
		top_widget = colony_item_card_at_province_select
		looking_for_a = province_definition
		source = actor
		source_flags = possible_colonial_charters
		target_flag = target
		cache_targets = yes #these are the same no matter where we're coming from
		name = "create_colonial_charter_select_province_definition"
		none_available_msg_key = "create_colonial_charter_no_provinces"
		column = {
			data = colonial_charter
		}
		#no trigger, possible_colonial_charters does this for us in code
		enabled = {
			scope:actor = {
				is_valid_colonial_charter = prev
				custom_tooltip = {
					text = settle_the_frontier_ct
					not = {
						any_cabinet = {
							has_cabinet_action = yes
							cabinet_action = cabinet_action:settle_the_frontier
							interaction_target:target = root
						}
					}
				}
			}
		}
	}

	select_trigger = {
		default_sort = population
		top_widget = colony_item_card_at_province_select
		looking_for_a = province
		source = actor
		target_flag = source
		cache_targets = yes #always the same set
		pre_evaluation_sort_value = {
			value = population
		}
		pre_evaluation_number_to_evaluate_fully = 5
		max_targets_for_ui = 100 #big countries really punish the UI
		name = "create_colonial_charter_select_province"
		none_available_msg_key = "create_colonial_charter_select_province_none"
		column = {
			data = create_colonial_charter_column
		}
		column = {
			data = population
		}
		enabled = {
			population > 1
		}
	}
 
	effect = {
		scope:actor = {
			if = {
				limit = {
					exists = scope:source
					exists = scope:target
				}
				create_colonial_charter = {
					source = scope:source
					target = scope:target
				}
			}
		}
	}

	ai_will_do = {
		if = {
			# only portugal can colonize brazil
			limit = { scope:target = { region = region:brazil_region } }
			if = {
				limit = {
					scope:actor = {
						or = {
							tag = POR
							and = {
								is_subject_of = c:POR
								is_colonial_subject = yes
							}
						}
					}
				}
				add = "scope:actor.colonial_charter_utility(scope:target|scope:source)"
				multiply = 2
			}
		}
		else_if = {
			# only spain can colonize hispanic america
			limit = {
				scope:target = {
					continent = continent:america
					region != region:east_coast_region
					region != region:canada_region
					region != region:great_plains_region
					region != region:alaska_region
				}
			}
			if = {
				limit = {
					scope:actor = {
						or = {
							tag = CAS
							is_subject_of = c:CAS
							tag = SPA
							is_subject_of = c:SPA
						}
					}
				}
				add = "scope:actor.colonial_charter_utility(scope:target|scope:source)"
				multiply = 2
			}
		}
		else_if = {
			# kilwa and zimbabwe cant colonize outside of their home regions
			limit = {
				scope:target = {
					region != region:swahili_coast_region
					region != region:zimbabwe_region
					or = {
						tag = ZAN # kilwa
						tag = ZIM # zimbabwe
					}
				}
			}
			add = "scope:actor.colonial_charter_utility(scope:target|scope:source)"
			divide = 2
		}
		else_if = {
			# kongo cant colonize outside of their home region
			limit = {
				scope:target = {
					region != region:kongo_region
					or = {
						tag = KON
						and = {
							capital ?= {
								region = region:kongo_region
							}
							religion = religion:bantu_religion
						}
					}
				}
			}
			add = "scope:actor.colonial_charter_utility(scope:target|scope:source)"
			divide = 2
		}
		else_if = {
			limit = {
				# country can colonize
				scope:actor = {
					or = {
						is_colonial_subject = yes
						# scandinavia
						tag = NOR
						tag = SWE
						tag = FIN
						# western
						tag = POR
						tag = CAS
						tag = SPA
						tag = ENG
						tag = GBR
						tag = FRA
						tag = NED
						# eastern
						tag = RUS
						# americans
						tag = MAY
						religion = religion:inti
						# asians #
						tag = JAP
					}
				}
			}
			add = "scope:actor.colonial_charter_utility(scope:target|scope:source)"
			if = {
				limit = {
					scope:target = {
						continent = continent:africa
					}
				}
				divide = 4
				if = {
					limit = {
						not = { any_location_in_province = { is_coastal = yes } }
					}
					subtract = 1000
				}
			}
		}
		else = {
			subtract = 1000
		}
	}
}
