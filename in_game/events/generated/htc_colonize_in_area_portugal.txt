namespace = htc_colonize_in_area_portugal

htc_colonize_in_area_portugal.1 = {
	type = country_event
	title = htc_colonize_in_area.title
	desc = htc_colonize_in_area.desc
	fire_only_once = no
	
	dynamic_historical_event = {
        tag = POR
		from = 1480.1.1
		to = 1520.1.1
		monthly_chance = 10
	}

	illustration_tags = {
		10 = happy
		10 = exterior
	}

	trigger = {
		area:gold_coast_area = {
			any_province_definition_in_area = {
				not = { has_colonial_charter = root }
				any_location_in_province_definition = {
					or = {
						is_adjacent_to_owned_or_subject_owns = yes
						within_colonial_range_of = root
					}
					has_owner = no
					is_ownable = yes
					is_discovered_by = root
				}
			}
		}
	}
	
	immediate = {
		area:gold_coast_area = {
			save_scope_as = source
			random_province_definition_in_area = {
				limit = {
					not = { has_colonial_charter = root }
					any_location_in_province_definition = {
						or = {
							is_adjacent_to_owned_or_subject_owns = yes
							within_colonial_range_of = root
						}
						has_owner = no
						is_ownable = yes
						is_discovered_by = root
					}
				}
				save_scope_as = geo
			}
		}
	}

	option = {
		name = htc.options.take
		historical_option = yes

		create_colonial_charter = {
			target = scope:geo
		}
	}
	option = {
		name = htc.options.no
	}
}
