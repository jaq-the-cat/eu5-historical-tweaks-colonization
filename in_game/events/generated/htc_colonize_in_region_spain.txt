namespace = htc_colonize_in_region_spain

htc_colonize_in_region_spain.1 = {
	type = country_event
	title = htc_colonize_in_region.title
	desc = htc_colonize_in_region.desc
	fire_only_once = no
	
	dynamic_historical_event = {
        tag = CAS
        tag = SPA
		from = 1505.1.1
		to = 1836.1.1
		monthly_chance = 5
	}

	illustration_tags = {
		10 = happy
		10 = exterior
	}

	trigger = {
		is_ai = yes
		monthly_balance > 25
		any_province_definition = {
			region = region:colombia_region
			not = { has_colonial_charter = root }
			not = {
				any_location_in_province_definition = {
					has_owner = yes
					owner = { is_ai = no }
				}
			}
			any_location_in_province_definition = {
				or = {
					is_adjacent_to_owned_or_subject_owns = yes
					within_colonial_range_of = root
				}
				has_owner = no
				is_ownable = yes
				is_discovered_by = root
			}
		}
	}
	
	immediate = {
		region:colombia_region = {
			save_scope_as = source
		}
		random_province_definition = {
			limit = {
				region = region:colombia_region
				not = { has_colonial_charter = root }
				not = {
					any_location_in_province_definition = {
    					has_owner = yes
    					owner = { is_ai = no }
					}
				}
				any_location_in_province_definition = {
					or = {
						is_adjacent_to_owned_or_subject_owns = yes
						within_colonial_range_of = root
					}
					has_owner = no
					is_ownable = yes
					is_discovered_by = root
				}
			}
			save_scope_as = geo
		}
	}

	option = {
		name = htc.options.take
		historical_option = yes

		create_colonial_charter = {
			target = scope:geo
		}
	}
	option = {
		name = htc.options.no
	}
}
