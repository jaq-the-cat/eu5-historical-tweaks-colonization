namespace = htc_colonize_province_portugal

htc_colonize_province_portugal.1 = {
	type = country_event
	title = htc_colonize_province.title
	desc = htc_colonize_province.desc
	fire_only_once = yes
	
	dynamic_historical_event = {
        tag = POR
		from = 1419.1.1
		to = 1500.1.1
		monthly_chance = 15
	}

	illustration_tags = {
		10 = happy
		10 = exterior
	}

	trigger = {
		monthly_balance > 25
		province_definition:madeira_province = {
			any_location_in_province_definition = {
				or = {
					is_adjacent_to_owned_or_subject_owns = yes
					within_colonial_range_of = root
				}
				has_owner = no
				is_ownable = yes
				is_discovered_by = root
			}
			not = { has_colonial_charter = root }
		}
	}

	immediate = {
		province_definition:madeira_province = {
			save_scope_as = geo
		}
	}
	

	option = {
		name = htc.options.take
		historical_option = yes

		create_colonial_charter = {
			target = province_definition:madeira_province
		}
	}
	option = {
		name = htc.options.no
	}
}
