namespace = htc_colonize_province_castille

htc_colonize_province_castille.1 = {
		type = country_event
		title = htc_colonize_province_castille.1.title
		desc = htc_colonize_province_castille.1.desc
		fire_only_once = yes
		
		dynamic_historical_event = {
            tag = CAS
            tag = SPA
			from = 1419.1.1
			to = 1500.1.1
			monthly_chance = 10
		}

		illustration_tags = {
			10 = happy
			10 = exterior
		}

		trigger = {
			province_definition:madeira_province = {
				any_location_in_province_definition = {
					has_owner = no
					is_discovered_by = root
					within_colonial_range_of = root
				}
				not = { has_colonial_charter = root }
				not = {
					any_present_country = {
						filter = { has_colonial_charters = yes }
						has_colonial_charter_in = prev
					}
				}
			}
		}

		immediate = {
			province_definition:madeira_province = {
				save_scope_as = geo
			}
		}
		

		option = {
			name = htc.options.take
			historical_option = yes

			create_colonial_charter = {
				target = province_definition:madeira_province
			}
		}
		option = {
			name = htc.options.no
		}
	}
