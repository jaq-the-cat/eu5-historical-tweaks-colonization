namespace = htc_take_location_portugal

htc_take_location_portugal.1 = {
		type = country_event
		title = htc_take_location_portugal.1.title
		desc = htc_take_location_portugal.1.desc
		fire_only_once = yes
		
		dynamic_historical_event = {
            tag = POR
			from = 1434.1.1
			to = 1500.1.1
			monthly_chance = 10
		}

		illustration_tags = {
			10 = happy
			10 = exterior
		}

		trigger = {
			location:bojador = {
				has_owner = no
				is_ownable = yes
				within_colonial_range_of = root
				is_discovered_by = root
			}
		}		

		immediate = {
			location:bojador = {
				save_scope_as = geo
			}
		}
		

		option = {
			name = htc.options.take
			historical_option = yes
			
			location:bojador = {
				change_location_owner = root
			}
		}
		option = {
			name = htc.options.no
		}
	}
